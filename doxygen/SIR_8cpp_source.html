64  std::pair<std::string, bool> compare(const DiffWriter& other) {
65 
66  std::size_t minSize = std::min(statements_.size(), other.getStatements().size());
67  if(minSize == 0 && (statements_.size() != other.getStatements().size()))
68  return std::make_pair("[AST mismatch] AST is empty", false);
69 
70  for(std::size_t idx = 0; idx < minSize; ++idx) {
71  if(!statements_[idx]->equals(other.getStatements()[idx].get())) {
72  return std::make_pair(
73  dawn::format("[AST mismatch] Statement mismatch\n"
74  " Actual:\n"
75  " %s\n"
76  " Expected:\n"
77  " %s",
78  indent(ASTStringifer::toString(statements_[idx]), 4),
79  indent(ASTStringifer::toString(other.getStatements()[idx]), 4)),
80  false);
81  }
82  }
83 
84  return std::make_pair("", true);
85  }
86 
87 private:
88  std::vector<std::shared_ptr<Stmt>> statements_;
89 };

108 std::pair<std::string, bool> compareAst(const std::shared_ptr<AST>& lhs,
109  const std::shared_ptr<AST>& rhs) {
110  if(lhs->getRoot()->equals(rhs->getRoot().get()))
111  return std::make_pair("", true);
112 
113  DiffWriter lhsDW, rhsDW;
114  lhs->accept(lhsDW);
115  rhs->accept(rhsDW);
116 
117  auto comp = lhsDW.compare(rhsDW);
118  if(!comp.second)
119  return comp;
120 
121  return std::make_pair("", true);
122 }

155 static std::pair<std::string, bool> pointerMapComparison(const sir::GlobalVariableMap& map1,
156  const sir::GlobalVariableMap& map2) {
157  std::string output;
158  if(map1.size() != map2.size()) {
159  output += dawn::format("[GlobalVariableMap mismatch] Number of Global Variables do not match\n"
160  " Actual:\n"
161  " %s\n"
162  " Expected:\n"
163  " %s",
164  map1.size(), map2.size());
165  return std::make_pair(output, false);
166 
167  } else {
168  for(auto& a : map1) {
169  auto finder = map2.find(a.first);
170  if(finder == map2.end()) {
171  output +=
172  dawn::format("[GlobalVariableMap mismatch] Global Variable '%s' not found\n", a.first);
173  return std::make_pair(output, false);
174  } else if(!(*(finder->second.get()) == *(a.second.get()))) {
175  output +=
176  dawn::format("[GlobalVariableMap mismatch] Global Variables '%s' values are not equal\n"
177  " Actual:\n"
178  " %s\n"
179  " Expected:\n"
180  " %s",
181  a.first, a.second->toString(), finder->second->toString());
182  return std::make_pair(output, false);
183  }
184  }
185  return std::make_pair(output, true);
186  }
187 }

191 std::pair<std::string, bool> SIR::comparison(const SIR& rhs) const {
192  std::string output;
193 
194  // Stencils
195  if((Stencils.size() != rhs.Stencils.size()))
196  return std::make_pair("[SIR mismatch] number of Stencils do not match\n", false);
197 
198  if(!std::equal(Stencils.begin(), Stencils.end(), rhs.Stencils.begin(),
199  pointeeComparison<sir::Stencil>)) {
200  output += "[SIR mismatch] Stencils do not match\n";
201  for(int i = 0; i < Stencils.size(); ++i) {
202  auto comp = pointeeComparisonWithOutput(Stencils[i], rhs.Stencils[i]);
203  if(comp.second == false) {
204  output += comp.first;
205  }
206  }
207 
208  return std::make_pair(output, false);
209  }
210 
211  // Stencil Functions
212  if(StencilFunctions.size() != rhs.StencilFunctions.size())
213  return std::make_pair("[SIR mismatch] number of Stencil Functions does not match\n", false);
214 
215  if(!std::equal(StencilFunctions.begin(), StencilFunctions.end(), rhs.StencilFunctions.begin(),
216  pointeeComparison<sir::StencilFunction>)) {
217  output += "[SIR mismatch] Stencil Functions do not match\n";
218  for(int i = 0; i < StencilFunctions.size(); ++i) {
219  auto comp = pointeeComparisonWithOutput(StencilFunctions[i], rhs.StencilFunctions[i]);
220  if(!comp.second)
221  output +=
222  dawn::format("[StencilFunction mismatch] Stencil Function %s does not match\n%s\n",
223  StencilFunctions[i]->Name, comp.first);
224  }
225 
226  return std::make_pair(output, false);
227  }
228 
229  // Global variable map
230  if(GlobalVariableMap.get()->size() != rhs.GlobalVariableMap.get()->size())
231  return std::make_pair("[SIR mismatch] number of Global Variables does not match\n", false);
232 
233  if(!pointerMapComparison(*(GlobalVariableMap.get()), *(rhs.GlobalVariableMap.get())).second) {
234  auto comp = pointerMapComparison(*(GlobalVariableMap.get()), *(rhs.GlobalVariableMap.get()));
235  if(!comp.second)
236  return std::make_pair(comp.first, false);
237  }
238 
239  return std::make_pair("", true);
240 }

242 std::pair<std::string, bool> sir::Stencil::comparison(const sir::Stencil& rhs) const {
243  // Fields
244  if(Fields.size() != rhs.Fields.size())
245  return std::make_pair(dawn::format("[Stencil mismatch] number of Fields does not match\n"
246  " Actual:\n"
247  " %s\n"
248  " Expected:\n"
249  " %s",
250  Fields.size(), rhs.Fields.size()),
251  false);
252 
253  // Name
254  if(Name != rhs.Name)
255  return std::make_pair(dawn::format("[Stencil mismatch] Stencil names do not match\n"
256  " Actual:\n"
257  " %s\n"
258  " Expected:\n"
259  " %s",
260  Name, rhs.Name),
261  false);
262 
263  // Attributes
264  if(Attributes != rhs.Attributes)
265  return std::make_pair(dawn::format("[Stencil mismatch] Stencil attibutes do not match\n"
266  " Actual:\n"
267  " %s\n"
268  " Expected:\n"
269  " %s",
270  Attributes.getBits(), rhs.Attributes.getBits()),
271  false);
272 
273  if(!Fields.empty() &&
274  !std::equal(Fields.begin(), Fields.end(), rhs.Fields.begin(), pointeeComparison<sir::Field>)) {
275  std::string output = "[Stencil mismatch] Fields do not match\n";
276 
277  for(int i = 0; i < Fields.size(); ++i) {
278  auto comp = pointeeComparisonWithOutput(Fields[i], rhs.Fields[i]);
279  if(!comp.second)
280  output += dawn::format("[Stencil mismatch] Field %s does not match\n%s\n",
281  Fields[i].get()->Name, comp.first);
282  }
283 
284  return std::make_pair(output, false);
285  }
286 
287  // AST
288  auto astComp = compareAst(StencilDescAst, rhs.StencilDescAst);
289  if(!astComp.second)
290  return std::make_pair(dawn::format("[Stencil mismatch] ASTs do not match\n%s\n", astComp.first),
291  false);
292 
293  return std::make_pair("", true);
294 }

296 std::pair<std::string, bool>
297 sir::StencilFunction::comparison(const sir::StencilFunction& rhs) const {
298 
299  // Name
300  if(Name != rhs.Name) {
301  return std::make_pair(
302  dawn::format("[StencilFunction mismatch] Names of Stencil Functions do not match\n"
303  " Actual:\n"
304  " %s\n"
305  " Expected:\n"
306  " %s",
307  Name, rhs.Name),
308  false);
309  }
310 
311  // Attributes
312  if(Attributes != rhs.Attributes) {
313  return std::make_pair(
314  dawn::format("[StencilFunction mismatch] Attributes of Stencil Functions do not match\n"
315  " Actual:\n"
316  " %i\n"
317  " Expected:\n"
318  " %i",
319  Attributes.getBits(), rhs.Attributes.getBits()),
320  false);
321  }
322 
323  // Arguments
324  if(Args.size() != rhs.Args.size()) {
325  return std::make_pair(
326  dawn::format("[StencilFunction mismatch] Number of Arguments do not match\n"
327  " Actual:\n"
328  " %i\n"
329  " Expected:\n"
330  " %i",
331  Args.size(), rhs.Args.size()),
332  false);
333  }
334 
335  if(!std::equal(Args.begin(), Args.end(), rhs.Args.begin(),
336  pointeeComparison<sir::StencilFunctionArg>)) {
337  std::string output = "[StencilFunction mismatch] Stencil Functions Arguments do not match\n";
338  for(int i = 0; i < Args.size(); ++i) {
339  auto comp = pointeeComparisonWithOutput(Args[i], rhs.Args[i]);
340  if(!comp.second) {
341  output += dawn::format("[StencilFunction mismatch] Argument '%s' does not match\n%s\n",
342  Args[i]->Name, comp.first);
343  }
344  }
345  return std::make_pair(output, false);
346  }
347 
348  // Intervals
349  if(Intervals.size() != rhs.Intervals.size()) {
350  return std::make_pair(
351  dawn::format("[StencilFunction mismatch] Number of Intervals do not match\n"
352  " Actual:\n"
353  " %i\n"
354  " Expected:\n"
355  " %i",
356  Intervals.size(), rhs.Intervals.size()),
357  false);
358  }
359 
360  // Intervals
361  if(!Intervals.empty() &&
362  !std::equal(Intervals.begin(), Intervals.end(), rhs.Intervals.begin(),
363  pointeeComparison<sir::Interval>)) {
364  std::string output = "[StencilFunction mismatch] Intervals do not match\n";
365  for(int i = 0; i < Intervals.size(); ++i) {
366  auto comp = pointeeComparisonWithOutput(Intervals[i], rhs.Intervals[i]);
367  if(comp.second == false) {
368  output += dawn::format("[StencilFunction mismatch] Interval '%s' does not match '%s'\n%s\n",
369  *Intervals[i], *rhs.Intervals[i], comp.first);
370  }
371  }
372  return std::make_pair(output, false);
373  }
374 
375  // ASTs
376  if(Asts.size() != rhs.Asts.size()) {
377  return std::make_pair(dawn::format("[StencilFunction mismatch] Number of ASTs does not match\n"
378  " Actual:\n"
379  " %i\n"
380  " Expected:\n"
381  " %i",
382  Asts.size(), rhs.Asts.size()),
383  false);
384  }
385 
386  auto intervalToString = [](const sir::Interval& interval) {
387  std::stringstream ss;
388  ss << interval;
389  return ss.str();
390  };
391 
392  for(int i = 0; i < Asts.size(); ++i) {
393  auto astComp = compareAst(Asts[i], rhs.Asts[i]);
394  if(!astComp.second)
395  return std::make_pair(
396  dawn::format("[StencilFunction mismatch] AST '%s' does not match\n%s\n",
397  i < Intervals.size() ? intervalToString(*Intervals[i]) : std::to_string(i),
398  astComp.first),
399  false);
400  }
401 
402  return std::make_pair("", true);
403 }

442 std::pair<std::string, bool> sir::Value::comparison(const sir::Value& rhs) const {
443  auto type = getType();
444  if(type != rhs.getType())
445  return std::make_pair(dawn::format("[Value mismatch] Values are not of the same type\n"
446  " Actual:\n"
447  " %s\n"
448  " Expected:\n"
449  " %s",
450  type, rhs.getType()),
451  false);
452 
453  switch(type) {
454  case sir::Value::TypeKind::Boolean:
455  return isEqualImpl<bool>(*this, rhs, rhs.toString());
456  case sir::Value::TypeKind::Integer:
457  return isEqualImpl<int>(*this, rhs, rhs.toString());
458  case sir::Value::TypeKind::Double:
459  return isEqualImpl<double>(*this, rhs, rhs.toString());
460  case sir::Value::TypeKind::String:
461  return isEqualImpl<std::string>(*this, rhs, rhs.toString());
462  default:
463  dawn_unreachable("invalid type");
464  }

581  os << indent2 << "Offset : " << offset->Name << "\n";

585  os << "\n" << indent2 << "Do\n";

604  os << indent1 << "}\n";

607  os << "\n}";
