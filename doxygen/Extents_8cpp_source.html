76 Extents::getVerticalLoopOrderAccesses(LoopOrderKind loopOrder) const {
77  VerticalLoopOrderAccess access{false, false};
78 
79  if(isVerticalPointwise())
80  return access;
81 
82  const Extent& verticalExtent = extents_[2];
83 
84  switch(loopOrder) {
85  case LoopOrderKind::LK_Parallel:
86  // Any accesses in the vertical are against the loop-order
87  return VerticalLoopOrderAccess{true, true};
88 
89  case LoopOrderKind::LK_Forward: {
90  // Accesses k+1 are against the loop order
91  if(verticalExtent.Plus > 0)
92  access.CounterLoopOrder = true;
93  if(verticalExtent.Minus < 0)
94  access.LoopOrder = true;
95  break;
96  }
97  case LoopOrderKind::LK_Backward:
98  // Accesses k-1 are against the loop order
99  if(verticalExtent.Minus < 0)
100  access.CounterLoopOrder = true;
101  if(verticalExtent.Plus > 0)
102  access.LoopOrder = true;
103  break;
104  }
105 
106  return access;
107 }
