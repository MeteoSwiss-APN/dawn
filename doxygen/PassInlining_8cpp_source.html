107  virtual void visit(const std::shared_ptr<ReturnStmt>& stmt) override {
108  DAWN_ASSERT_MSG(scopeDepth_ == 1, "cannot inline nested return statement!");
109 
110  // Instead of returning a value, we assign it to a local variable
111  if(AccessIDOfCaller_ == 0) {
112  // We are *not* called within an arugment list of a stencil function, meaning we can store the
113  // return value in a local variable.
114  int AccessID = instantiation_->nextUID();
115  auto returnVarName = StencilInstantiation::makeLocalVariablename(
116  curStencilFunctioninstantiation_->getName(), AccessID);
117 
118  newExpr_ = std::make_shared<VarAccessExpr>(returnVarName);
119  auto newStmt = std::make_shared<VarDeclStmt>(
120  dawn::Type(BuiltinTypeID::Float, CVQualifier::Const), returnVarName, 0, "=",
121  std::vector<std::shared_ptr<Expr>>{stmt->getExpr()});
122  newStmtAccessesPairs_.emplace_back(std::make_shared<StatementAccessesPair>(
123  std::make_shared<Statement>(newStmt, oldStmtAccessesPair_->getStatement()->StackTrace)));
124 
125  // Register the variable
126  instantiation_->setAccessIDNamePair(AccessID, returnVarName);
127  instantiation_->getStmtToAccessIDMap().emplace(newStmt, AccessID);
128  instantiation_->getExprToAccessIDMap().emplace(newExpr_, AccessID);
129 
130  } else {
131  // We are called within an arugment list of a stencil function, we thus need to store the
132  // return value in temporary storage (we only land here if we do precomputations).
133  auto returnFieldName = StencilInstantiation::makeTemporaryFieldname(
134  curStencilFunctioninstantiation_->getName(), AccessIDOfCaller_);
135 
136  newExpr_ = std::make_shared<FieldAccessExpr>(returnFieldName);
137  auto newStmt =
138  std::make_shared<ExprStmt>(std::make_shared<AssignmentExpr>(newExpr_, stmt->getExpr()));
139  newStmtAccessesPairs_.emplace_back(std::make_shared<StatementAccessesPair>(
140  std::make_shared<Statement>(newStmt, oldStmtAccessesPair_->getStatement()->StackTrace)));
141 
142  // Promote the "temporary" storage we used to mock the argument to an actual temporary field
143  instantiation_->setAccessIDNamePairOfField(AccessIDOfCaller_, returnFieldName, true);
144  instantiation_->getExprToAccessIDMap().emplace(newExpr_, AccessIDOfCaller_);
145  }
146 
147  // Resolve the actual expression of the return statement
148  stmt->getExpr()->accept(*this);

203  void visit(const std::shared_ptr<StencilFunCallExpr>& expr) override {
204  // This is a nested stencil function call (i.e a stencil function call within the current
205  // stencil function)
206  StencilFunctionInstantiation* func =
207  curStencilFunctioninstantiation_->getStencilFunctionInstantiation(expr);
208  auto curStencilFunStmtAccessPair = newStmtAccessesPairs_[newStmtAccessesPairs_.size() - 1];
209 
210  int AccessIDOfCaller = 0;
211  if(!argListScope_.empty()) {
212  int argIdx = argListScope_.top().ArgumentIndex;
213  StencilFunctionInstantiation* curFunc = argListScope_.top().Function;
214 
215  // This stencil function is called within the argument list of another stencil function. Get
216  // the AccessID of the "temporary" storage we used to mock the argument (this will be
217  // converted to an actual temporary storage if we inline the function)
218  AccessIDOfCaller = curFunc->getCallerAccessIDOfArgField(argIdx);
219  DAWN_ASSERT(curFunc->isProvidedByStencilFunctionCall(AccessIDOfCaller));
220  }
221 
222  // Resolve the arguments
223  argListScope_.push(ArgListScope(func));
224  for(const auto& arg : expr->getArguments())
225  arg->accept(*this);
226  argListScope_.pop();
227 
228  // Try to inline the stencil-function
229  auto inlineResult = tryInlineStencilFunction(strategy_, func, oldStmtAccessesPair_,
230  newStmtAccessesPairs_, AccessIDOfCaller);
231  if(inlineResult.first) {
232 
233  // Compute the index of the statement of our current stencil-function call
234  auto curStencilFunStmtIt = std::find(
235  newStmtAccessesPairs_.begin(), newStmtAccessesPairs_.end(), curStencilFunStmtAccessPair);
236  DAWN_ASSERT(curStencilFunStmtIt != newStmtAccessesPairs_.end());
237  int stmtIdxOfFunc = std::distance(newStmtAccessesPairs_.begin(), curStencilFunStmtIt);
238 
239  if(func->hasReturn()) {
240  std::unique_ptr<Inliner>& inliner = inlineResult.second;
241  DAWN_ASSERT(inliner);
242  DAWN_ASSERT(inliner->getNewExpr());
243 
244  // We need to change the current statement s.t instead of calling the stencil-function it
245  // accesses the precomputed value. In addition the statement needs to be last again (we
246  // push backed all the new statements). Hence, we need to insert an empty statement in the
247  // back -> swap with our statement -> replace the expr in our statement and evict the empty
248  // statement)
249  newStmtAccessesPairs_.emplace_back(
250  std::make_shared<StatementAccessesPair>(std::make_shared<Statement>(nullptr, nullptr)));
251  std::iter_swap(newStmtAccessesPairs_.begin() + stmtIdxOfFunc,
252  std::prev(newStmtAccessesPairs_.end()));
253 
254  replaceOldExprWithNewExprInStmt(
255  newStmtAccessesPairs_[newStmtAccessesPairs_.size() - 1]->getStatement()->ASTStmt, expr,
256  inliner->getNewExpr());
257  }
258 
259  // Erase the statement of the original stencil function call. The statment is either empty
260  // (in case it had a return value) or it just contains the function call which we inlined.
261  newStmtAccessesPairs_.erase(newStmtAccessesPairs_.begin() + stmtIdxOfFunc);
262 
263  // Remove the function
264  instantiation_->removeStencilFunctionInstantiation(expr, curStencilFunctioninstantiation_);
265 
266  } else {
267  // Inlining failed, transfer ownership
268  instantiation_->getExprToStencilFunctionInstantiationMap().emplace(expr, func);
269  }
270 
271  if(!argListScope_.empty())
272  argListScope_.top().ArgumentIndex++;
273  }

402  // of the stencil function call

405 
406  // Remove the stencil-function (`nullptr` means we don't have a nested stencil function)
407  instantiation_->removeStencilFunctionInstantiation(expr, nullptr);
408  }
409 
410  if(!argListScope_.empty())
411  argListScope_.top().ArgumentIndex++;
412  }

439 static std::pair<bool, std::unique_ptr<Inliner>>
440 tryInlineStencilFunction(PassInlining::InlineStrategyKind strategy,
441  StencilFunctionInstantiation* stencilFunc,
442  std::shared_ptr<StatementAccessesPair>& oldStmtAccessesPair,
443  std::vector<std::shared_ptr<StatementAccessesPair>>& newStmtAccessesPairs,
444  int AccessIDOfCaller) {
445 
446  // Function which do not return a value are *always* inlined. Function which do return a value
447  // are only inlined if we favor precomputations.
448  if(!stencilFunc->hasReturn() || strategy == PassInlining::IK_Precomputation) {
449  auto inliner = make_unique<Inliner>(strategy, stencilFunc, oldStmtAccessesPair,
450  newStmtAccessesPairs, AccessIDOfCaller);
451  stencilFunc->getAST()->accept(*inliner);
452  return std::pair<bool, std::unique_ptr<Inliner>>(true, std::move(inliner));
453  }
454  return std::pair<bool, std::unique_ptr<Inliner>>(false, nullptr);
455 }

462 bool PassInlining::run(StencilInstantiation* stencilInstantiation) {
463  // Nothing to do ...
464  if(strategy_ == IK_None)
465  return true;
466 
467  DetectInlineCandiates inliner(strategy_, stencilInstantiation);
468 
469  // Iterate all statements (top -> bottom)
470  for(auto& stencilPtr : stencilInstantiation->getStencils()) {
471  for(auto& multiStagePtr : stencilPtr->getMultiStages()) {
472  for(auto& stagePtr : multiStagePtr->getStages()) {
473  Stage& stage = *stagePtr;
474  DoMethod& doMethod = stage.getSingleDoMethod();
475 
476  auto& stmtAccList = doMethod.getStatementAccessesPairs();
477  auto stmtAccIt = stmtAccList.begin();
478 
479  for(; stmtAccIt != stmtAccList.end(); ++stmtAccIt) {
480  inliner.processStatment(*stmtAccIt);
481 
482  if(inliner.inlineCandiatesFound()) {
483  const auto& newStmtAccList = inliner.getNewStatementAccessesPairs();
484 
485  // Compute the accesses of the new statements
486  computeAccesses(stencilInstantiation, newStmtAccList);
487 
488  // Erase the old StatementAccessPair ...
489  stmtAccIt = stmtAccList.erase(stmtAccIt);
490 
491  // ... and insert the new ones
492  stmtAccIt = stmtAccList.insert(stmtAccIt, newStmtAccList.begin(), newStmtAccList.end());
493  std::advance(stmtAccIt, newStmtAccList.size() - 1);
494  }
495  }
496 
497  stage.update();
498  }
499  }
500  }
501  return true;
502 }
